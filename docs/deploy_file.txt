name: Zeros1 CI/CD Deploy

on: 
    push: 
        branches: 
            - main

jobs:
  deploy:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout code
        uses: actions/checkout@v3

      - name: Set up Docker
        uses: docker/setup-buildx-action@v2

      - name: Log in to Docker Hub
        uses: docker/login-action@v2
        with:
          username: ${{ secrets.DOCKER_USERNAME }}
          password: ${{ secrets.DOCKER_PASSWORD }}

      - name: Build Docker images
        run: |
          docker build -t my-frontend:latest ./frontend
          docker build -t my-backend:latest ./backend

      - name: Push images (optional)
        run: |
          docker tag my-frontend:latest yourdockerhubuser/my-frontend:latest
          docker tag my-backend:latest yourdockerhubuser/my-backend:latest
          docker push yourdockerhubuser/my-frontend:latest
          docker push yourdockerhubuser/my-backend:latest

      - name: SSH and deploy
        uses: appleboy/ssh-action@v0.1.7
        with:
          host: ${{ secrets.SERVER_IP }}
          username: ${{ secrets.SERVER_USER }}
          key: ${{ secrets.ZEROS_CI_CD_SECRET }}
          script: |
            cd /path/to/project
            git pull origin main
            docker compose build
            docker compose up -d
